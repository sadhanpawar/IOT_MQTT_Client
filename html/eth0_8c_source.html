257 // Initializes ethernet device
258 // Uses order suggested in Chapter 6 of datasheet except 6.4 OST which is first here
259 void initEther(uint16_t mode)
260 {
261  // Initialize SPI0
262  initSpi0(USE_SSI0_RX);
263  setSpi0BaudRate(10e6, 40e6);
264  setSpi0Mode(0, 0);
265 
266  // Enable clocks
267  enablePort(PORTA);
268  enablePort(PORTB);
269  enablePort(PORTC);
270 
271  // Configure pins for ethernet module
272  selectPinPushPullOutput(CS);
273  selectPinDigitalInput(WOL);
274  selectPinDigitalInput(INT);
275 
276  // make sure that oscillator start-up timer has expired
277  while ((readEtherReg(ESTAT) & CLKRDY) == 0) {}
278 
279  // disable transmission and reception of packets
280  clearEtherReg(ECON1, RXEN);
281  clearEtherReg(ECON1, TXRTS);
282 
283  // initialize receive buffer space
284  setEtherBank(ERXSTL);
285  writeEtherReg(ERXSTL, LOBYTE(0x0000));
286  writeEtherReg(ERXSTH, HIBYTE(0x0000));
287  writeEtherReg(ERXNDL, LOBYTE(0x1A09));
288  writeEtherReg(ERXNDH, HIBYTE(0x1A09));
289 
290  // initialize receiver write and read ptrs
291  // at startup, will write from 0 to 1A08 only and will not overwrite rd ptr
292  writeEtherReg(ERXWRPTL, LOBYTE(0x0000));
293  writeEtherReg(ERXWRPTH, HIBYTE(0x0000));
294  writeEtherReg(ERXRDPTL, LOBYTE(0x1A09));
295  writeEtherReg(ERXRDPTH, HIBYTE(0x1A09));
296  writeEtherReg(ERDPTL, LOBYTE(0x0000));
297  writeEtherReg(ERDPTH, HIBYTE(0x0000));
298 
299  // setup receive filter
300  // always check CRC, use OR mode
301  setEtherBank(ERXFCON);
302  writeEtherReg(ERXFCON, (mode | ETHER_CHECKCRC) & 0xFF);
303 
304  // bring mac out of reset
305  setEtherBank(MACON2);
306  writeEtherReg(MACON2, 0);
307 
308  // enable mac rx, enable pause control for full duplex
309  writeEtherReg(MACON1, TXPAUS | RXPAUS | MARXEN);
310 
311  // enable padding to 60 bytes (no runt packets)
312  // add crc to tx packets, set full or half duplex
313  if ((mode & ETHER_FULLDUPLEX) != 0)
314  writeEtherReg(MACON3, FULDPX | FRMLNEN | TXCRCEN | PAD60);
315  else
316  writeEtherReg(MACON3, FRMLNEN | TXCRCEN | PAD60);
317 
318  // leave MACON4 as reset
319 
320  // set maximum rx packet size
321  writeEtherReg(MAMXFLL, LOBYTE(1518));
322  writeEtherReg(MAMXFLH, HIBYTE(1518));
323 
324  // set back-to-back inter-packet gap to 9.6us
325  if ((mode & ETHER_FULLDUPLEX) != 0)
326  writeEtherReg(MABBIPG, 0x15);
327  else
328  writeEtherReg(MABBIPG, 0x12);
329 
330  // set non-back-to-back inter-packet gap registers
331  writeEtherReg(MAIPGL, 0x12);
332  writeEtherReg(MAIPGH, 0x0C);
333 
334  // leave collision window MACLCON2 as reset
335 
336  // initialize phy duplex
337  if ((mode & ETHER_FULLDUPLEX) != 0)
338  writeEtherPhy(PHCON1, PDPXMD);
339  else
340  writeEtherPhy(PHCON1, 0);
341 
342  // disable phy loopback if in half-duplex mode
343  writeEtherPhy(PHCON2, HDLDIS);
344 
345  // Flash LEDA and LEDB
346  writeEtherPhy(PHLCON, 0x0880);
347  waitMicrosecond(100000);
348 
349  // set LEDA (link status) and LEDB (tx/rx activity)
350  // stretch LED on to 40ms (default)
351  writeEtherPhy(PHLCON, 0x0472);
352 
353  // enable reception
354  setEtherReg(ECON1, RXEN);
355 }

379 // Returns up to max_size characters in data buffer
380 // Returns number of bytes copied to buffer
381 // Contents written are 16-bit size, 16-bit status, payload excl crc
382 uint16_t getEtherPacket(etherHeader *ether, uint16_t maxSize)
383 {
384  uint16_t i = 0, size, tmp16, status;
385  uint8_t *packet = (uint8_t*)ether;
386 
387  // enable read from FIFO buffers
388  startEtherMemRead();
389 
390  // get next packet information
391  nextPacketLsb = readEtherMem();
392  nextPacketMsb = readEtherMem();
393 
394  // calc size
395  // don't return crc, instead return size + status, so size is correct
396  size = readEtherMem();
397  tmp16 = readEtherMem();
398  size |= (tmp16 << 8);
399 
400  // get status (currently unused)
401  status = readEtherMem();
402  tmp16 = readEtherMem();
403  status |= (tmp16 << 8);
404 
405  // copy data
406  if (size > maxSize)
407  size = maxSize;
408  while (i < size)
409  packet[i++] = readEtherMem();
410 
411  // end read from FIFO buffers
412  stopEtherMemRead();
413 
414  // advance read pointer
415  setEtherBank(ERXRDPTL);
416  writeEtherReg(ERXRDPTL, nextPacketLsb); // hw ptr
417  writeEtherReg(ERXRDPTH, nextPacketMsb);
418  writeEtherReg(ERDPTL, nextPacketLsb); // dma rd ptr
419  writeEtherReg(ERDPTH, nextPacketMsb);
420 
421  // decrement packet counter so that PKTIF is maintained correctly
422  setEtherReg(ECON2, PKTDEC);
423 
424  return size;
425 }

427 // Writes a packet
428 bool putEtherPacket(etherHeader *ether, uint16_t size)
429 {
430  uint16_t i;
431  uint8_t *packet = (uint8_t*) ether;
432 
433  // clear out any tx errors
434  if ((readEtherReg(EIR) & TXERIF) != 0)
435  {
436  clearEtherReg(EIR, TXERIF);
437  setEtherReg(ECON1, TXRTS);
438  clearEtherReg(ECON1, TXRTS);
439  }
440 
441  // set DMA start address
442  setEtherBank(EWRPTL);
443  writeEtherReg(EWRPTL, LOBYTE(0x1A0A));
444  writeEtherReg(EWRPTH, HIBYTE(0x1A0A));
445 
446  // start FIFO buffer write
447  startEtherMemWrite();
448 
449  // write control byte
450  writeEtherMem(0);
451 
452  // write data
453  for (i = 0; i < size; i++)
454  writeEtherMem(packet[i]);
455 
456  // stop write
457  stopEtherMemWrite();
458 
459  // request transmit
460  writeEtherReg(ETXSTL, LOBYTE(0x1A0A));
461  writeEtherReg(ETXSTH, HIBYTE(0x1A0A));
462  writeEtherReg(ETXNDL, LOBYTE(0x1A0A+size));
463  writeEtherReg(ETXNDH, HIBYTE(0x1A0A+size));
464  clearEtherReg(EIR, TXIF);
465  setEtherReg(ECON1, TXRTS);
466 
467  // wait for completion
468  while ((readEtherReg(ECON1) & TXRTS) != 0);
469 
470  // determine success
471  return ((readEtherReg(ESTAT) & TXABORT) == 0);
472 }
