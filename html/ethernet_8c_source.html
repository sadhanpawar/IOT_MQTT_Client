436  while (true)

488  }
489 
490  // Handle TCP datagram
491  if (isTcp(data))
492  {
493  tcpHandlerRx(data);
494  }
495  }
496  }
497  }
498 
499  /* for transmission */
500  mqttHandler(TransBuffer);
501  tcpHandlerTx(TransBuffer);
502  ipHandlerTx(TransBuffer);
503 
504  memset(buffer_rx,0,sizeof(buffer_rx));
505 
506  }
507 }
