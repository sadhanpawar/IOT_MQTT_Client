203 void setPinAuxFunction(PORT port, uint8_t pin, uint32_t fn)
204 {
205  // call with header file shifted values or 4-bit number
206  if (fn <= 15)
207  fn = fn << (pin*4);
208  else
209  fn = fn & (0x0000000F << (pin*4));
210  switch(port)
211  {
212  case PORTA:
213  GPIO_PORTA_PCTL_R = (GPIO_PORTA_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
214  break;
215  case PORTB:
216  GPIO_PORTB_PCTL_R = (GPIO_PORTB_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
217  break;
218  case PORTC:
219  GPIO_PORTC_PCTL_R = (GPIO_PORTC_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
220  break;
221  case PORTD:
222  GPIO_PORTD_PCTL_R = (GPIO_PORTD_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
223  break;
224  case PORTE:
225  GPIO_PORTE_PCTL_R = (GPIO_PORTE_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
226  break;
227  case PORTF:
228  GPIO_PORTF_PCTL_R = (GPIO_PORTF_PCTL_R & ~(0x0000000F << (pin*4))) | fn;
229  }
230  // set AFSEL bit only if using aux function, otherwise clear bit
231  uint32_t* p;
232  p = (uint32_t*)port + pin + OFS_DATA_TO_AFSEL;
233  *p = (fn > 0);
234 }
