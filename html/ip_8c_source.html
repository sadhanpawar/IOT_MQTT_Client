177 // Calculate sum of words
178 // Must use getEtherChecksum to complete 1's compliment addition
179 void sumIpWords(void* data, uint16_t sizeInBytes, uint32_t* sum)
180 {
181  uint8_t* pData = (uint8_t*)data;
182  uint16_t i;
183  uint8_t phase = 0;
184  uint16_t data_temp;
185  for (i = 0; i < sizeInBytes; i++)
186  {
187  if (phase)
188  {
189  data_temp = *pData;
190  *sum += data_temp << 8;
191  }
192  else
193  *sum += *pData;
194  phase = 1 - phase;
195  pData++;
196  }
197 }

199 // Completes 1's compliment addition by folding carries back into field
200 uint16_t getIpChecksum(uint32_t sum)
201 {
202  uint16_t result;
203  // this is based on rfc1071
204  while ((sum >> 16) > 0)
205  sum = (sum & 0xFFFF) + (sum >> 16);
206  result = sum & 0xFFFF;
207  return ~result;
208 }
