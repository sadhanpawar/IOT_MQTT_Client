91 void mqttPublish(etherHeader *ether, uint8_t *data, uint16_t size)
92 {
93  uint8_t *varHdrPtr;
94  uint8_t i,j;
95  uint16_t varLen = 0;
96  uint8_t idx = 0;
97  uint32_t encodedRemLen = 4 + mqttMcb.topicSize
98  + (mqttMcb.totalSize - mqttMcb.topicSize);
99 
100  mqttHeader *mqtt = (mqttHeader*)mqttMcb.data;
101  mqtt->controlPacket = MQ_PUBLISH;
102  mqtt->dup = 0; /*first try*/
103  mqtt->qosLevel = MQ_ACK_DELIVERY;
104  mqtt->retain = 1;
105 
106  mqttCalcDynLength(&mqtt->remLength,encodedRemLen,&idx);
107  if (idx > 0 ) idx -= 1;
108 
109  /*variable header*/
110 
111  /*topic length*/
112  varHdrPtr = mqtt->data;
113  *(varHdrPtr + idx + 0) = (mqttMcb.topicSize & 0xFF00) >> 8;
114  *(varHdrPtr + idx + 1) = (uint8_t)(mqttMcb.topicSize & 0x00FF);
115  varLen += 2;
116 
117  varHdrPtr = varHdrPtr + idx + (uint8_t)2;
118 
119  /*topic name*/
120  for(i = 0 ; i < mqttMcb.topicSize; i ++) {
121  varHdrPtr[i] = mqttMcb.args[i];
122  }
123  varLen += mqttMcb.topicSize;
124  varHdrPtr += i;
125 
126  /*Packet ID*/
127  *(varHdrPtr + 0)= (mqttPktId & 0xFF00) >> 8;
128  *(varHdrPtr+ 1) = (uint8_t)(mqttPktId & 0x00FF);
129 
130  varHdrPtr += 2;
131  varLen += 2;
132 
133  /*data*/
134  for(i = mqttMcb.topicSize,j=0; i < mqttMcb.topicSize + (mqttMcb.totalSize - mqttMcb.topicSize) ; i++,j++) {
135  varHdrPtr[j] = mqttMcb.args[i];
136  }
137 
138  varLen += (mqttMcb.totalSize - mqttMcb.topicSize);
139 
140  /*message length*/
141  //mqtt->remLength = varLen;
142  mqttMcb.totalSize = sizeof(mqttHeader) + varLen + idx;
143 
144  mqttSetTxStatus(true);
145 }

322  /*client ID length*/

324  *(varHdrPtr + idx + 11) = 0x17;
325  varLen += 2;
326 
327  /*client ID */

346 void mqttCalcDynLength(uint8_t *ptr, uint32_t enclen, uint8_t *noOfBytes)
347 {
348  uint8_t byte = 0;
349  uint8_t idx = 0;
350 
351  do {
352  byte = enclen % 128;
353  enclen /= 128;
354  if(enclen > 0) {
355  byte |= 0x80;
356  }
357  *ptr = byte;
358  ++ptr;
359  ++idx;
360  } while(enclen > 0);
361 
362  *noOfBytes = idx;
363 }
