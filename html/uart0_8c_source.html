59 void setUart0BaudRate(uint32_t baudRate, uint32_t fcyc)
60 {
61  uint32_t divisorTimes128 = (fcyc * 8) / baudRate; // calculate divisor (r) in units of 1/128,

63  divisorTimes128 += 1; // add 1/128 to allow rounding
64  UART0_CTL_R = 0; // turn-off UART0 to allow safe programming
65  UART0_IBRD_R = divisorTimes128 >> 7; // set integer value to floor(r)
66  UART0_FBRD_R = ((divisorTimes128) >> 1) & 63; // set fractional value to round(fract(r)*64)

68  UART0_CTL_R = UART_CTL_TXE | UART_CTL_RXE | UART_CTL_UARTEN;
